#Ez a ciklus addig fog ismétlődni, amég az állítás igaz nem lesz
while True:
    valasz = input('Írj END -et a program befejezéséhez')
    if valasz == 'END':
        break #A break parancs megszakítja a ciklust
while True:
    valasz = input('Írj END -et a program befejezéséhez')
    if valasz != 'END':
        continue #A continue parancs visszaugrik a ciklus elejére
    break
