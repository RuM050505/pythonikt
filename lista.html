Listák

#Listát így lehet létrehozni
lista = [] #Ez egy üres lista
lista = [10, 6, 23] #Ez a lista 3 integert tartalmaz
lista2 = ['cseresznye', 'alma', 'körte'] #Ez a lista 3 stringet tartalmaz

#Kiírja a lista összes elemét
print(lista2)

#Megszámolja hány elem van a listában
print(len(lista2))

#Kiírja a lista második (1) elemét
print(lista2[1])

#A - jel a lista legutolsó elemétől visszaszámol
print(lista2[-1])
#Üres listához elem adása
varosok = []
while True:
    varos = input('Adj meg városokat, hogy befejezd, nyomj ENTER-t')
    if varos == '':
        break
    varosok.append(varos) #A .append() a listához ad egy változót
print(varosok)
